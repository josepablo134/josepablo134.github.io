Example of use:

#define UHEAP_SIZE N
unsigned char heap_bytes[ UHEAP_SIZE ];
struct uheap* heap;
heap = uheap_init( heap_bytes, UHEAP_SIZE );
float* a = uheap_malloc( heap , sizeof(float) );
float* b = uheap_malloc( heap , sizeof(float) );
float* c = uheap_malloc( heap , sizeof(float) );
*a = 2.0;
*b = 1.0;
*c = *a + *b;
